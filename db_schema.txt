CREATE TABLE users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL
);

CREATE TABLE tasks (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    description TEXT,
    assignee_id INTEGER,
    status TEXT DEFAULT 'open',
    priority TEXT DEFAULT 'medium',
    deadline TEXT,  -- NEW: stores ISO date format
    created_at TEXT,
    updated_at TEXT,
    FOREIGN KEY (assignee_id) REFERENCES users (id)
);
